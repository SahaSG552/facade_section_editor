 * Calculate hinge positions for a door panel
 * @param {number} doorHeight - Door height in millimeters
 * @param {number} numHinges - Number of hinges (default: 2)
 * @param {number} offset - Distance from edge in millimeters (default: 20)
 * @returns {Array<Object>} Array of hinge positions with y coordinate and offset
 * @example
 * calculateHingePositions(2000, 3, 20)
 * // Returns: [{ y: 500, offset: 20 }, { y: 1000, offset: 20 }, { y: 1500, offset: 20 }]

  const positions = [];
  const spacing = doorHeight / (numHinges + 1);
  
  for (let i = 1; i <= numHinges; i++) {
